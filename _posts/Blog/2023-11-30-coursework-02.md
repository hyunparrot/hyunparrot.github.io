package com.example.test

fun main() {

    println("계산기 만들기")
    var operator = readLine() ?: '+'
    println("숫자를 입력해주세요.")
    var n1 = readLine()?.toInt() ?: 0
    println("숫자를 입력해주세요.")
    var n2 = readLine()?.toInt() ?: 0


    when (operator) {
        "+" -> {
            val result = n1 + n2
            println("$result")

        }

        "-" -> {
            var result = n1 - n2
            println("$result")

        }

        "*" -> {
            var result = n1 * n2
            println("$result")

        }

        '/' -> {
            var result = n1 / n2
            println("$result")
        }

        else -> {
            // 잘못된 연산자 일 때
            // 끝내거나 while문 쓰거나
            println("")
        }
